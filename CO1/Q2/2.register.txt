package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.View;
import android.widget.EditText;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
    }
    public void btnclick(View view){
        EditText edt = findViewById(R.id.name);
        EditText edt1 = findViewById(R.id.Address);
        EditText edt2 = findViewById(R.id.Email );
        TextView txt1 =findViewById(R.id.textView);
        TextView txt2 =findViewById(R.id.textView2 );
        TextView txt3 =findViewById(R.id.textView3);
        txt1.setText("hello"+edt .getText().toString() );
        txt2.setText("address:"+edt1 .getText().toString() );
        txt3.setText("email:"+edt2 .getText().toString() );
    }
}
